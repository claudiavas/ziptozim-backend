/**
 * Module to create ZIM files using zimwriterfs.
 * @module zimWriters
 */

const { spawn } = require('child_process');
const path = require('path');

// Path to the zimwriterfs executable
let zimwriterfsPath = path.join(__dirname, 'zimwriterfs');

/**
 * Creates a ZIM file from the specified source directory.
 * 
 * @param {string} sourceDirectory - The directory containing the files to be included in the ZIM file.
 * @param {string} zimFilePath - The path where the ZIM file will be saved.
 * @param {string} welcomePage - The path to the welcome page relative to the sourceDirectory.
 * @param {string} illustration - The path to the illustration image relative to the sourceDirectory.
 * @param {string} language - The language of the content in the ZIM file.
 * @param {string} title - The title of the ZIM file.
 * @param {string} description - A description of the ZIM file content.
 * @param {string} creator - The creator of the ZIM file content.
 * @param {string} publisher - The publisher of the ZIM file.
 * @returns {Promise<void>} A promise that resolves if the ZIM file is created successfully, or rejects with an error.
 */
exports.createZimFile = function(sourceDirectory, zimFilePath, welcomePage, illustration, language, title, description, creator, publisher) {
    return new Promise((resolve, reject) => {
        // Command arguments for zimwriterfs
        const command = [`--welcome=${welcomePage}`, `--illustration=${illustration}`, `--language=${language}`, `--title="${title}"`, `--description="${description}"`, `--creator=${creator}`, `--publisher=${publisher}`, sourceDirectory, zimFilePath];
        console.log("Zimwriterfs command", command)
        
        // Spawn the zimwriterfs process
        const zimwriterfs = spawn(zimwriterfsPath, command);
        
        // Log stdout data
        zimwriterfs.stdout.on('data', (data) => {
            console.log(`stdout: ${data}`);
        });

        // Log stderr data
        zimwriterfs.stderr.on('data', (data) => {
            console.error(`stderr: ${data}`);
        });

        // Handle process exit
        zimwriterfs.on('close', (code) => {
            console.log(`child process exited with code ${code}`);
            if (code !== 0) {
                reject(new Error(`zimwriterfs exited with code ${code}`));
            } else {
                resolve();
            }
        });
    });
}
